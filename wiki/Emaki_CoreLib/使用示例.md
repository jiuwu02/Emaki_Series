### 场景

创建一个简单的确认对话框 GUI。

### Java 代码

```java
import emaki.jiuwu.craft.corelib.gui.*;
import org.bukkit.configuration.file.YamlConfiguration;
import org.bukkit.entity.Player;
import org.bukkit.event.inventory.InventoryClickEvent;
import org.bukkit.event.inventory.InventoryCloseEvent;

import java.io.File;
import java.util.Map;
import java.util.function.Consumer;

public class ConfirmGui {
    
    private final GuiService guiService;
    private final GuiTemplate template;
    
    public ConfirmGui(JavaPlugin plugin, GuiService guiService) {
        this.guiService = guiService;
        
        File file = new File(plugin.getDataFolder(), "guis/confirm_gui.yml");
        YamlConfiguration config = YamlConfiguration.loadConfiguration(file);
        this.template = GuiTemplateParser.parse(config);
    }
    
    public void open(Player player, String message, Runnable onConfirm, Runnable onCancel) {
        GuiOpenRequest request = new GuiOpenRequest(
            plugin,
            player,
            template,
            Map.of("message", message),
            source -> null,  // 不需要物品工厂
            null,            // 使用默认渲染
            new ConfirmHandler(onConfirm, onCancel)
        );
        
        guiService.open(request);
    }
    
    private static class ConfirmHandler implements GuiSessionHandler {
        private final Runnable onConfirm;
        private final Runnable onCancel;
        
        public ConfirmHandler(Runnable onConfirm, Runnable onCancel) {
            this.onConfirm = onConfirm;
            this.onCancel = onCancel;
        }
        
        @Override
        public void onSlotClick(GuiSession session, InventoryClickEvent event, GuiTemplate.ResolvedSlot slot) {
            if (slot == null) return;
            
            String action = slot.definition().action();
            Player player = session.viewer();
            
            switch (action) {
                case "confirm" -> {
                    player.closeInventory();
                    if (onConfirm != null) {
                        onConfirm.run();
                    }
                }
                case "cancel" -> {
                    player.closeInventory();
                    if (onCancel != null) {
                        onCancel.run();
                    }
                }
            }
        }
        
        @Override
        public void onClose(GuiSession session, InventoryCloseEvent event) {
            // 可选：处理关闭事件
        }
    }
}

// 使用示例
ConfirmGui confirmGui = new ConfirmGui(plugin, guiService);
confirmGui.open(player, "确定要删除此物品吗？", 
    () -> {
        player.sendMessage("§a物品已删除");
        deleteItem();
    },
    () -> {
        player.sendMessage("§c操作已取消");
    }
);
```

```java
import emaki.jiuwu.craft.corelib.gui.*;
import org.bukkit.entity.Player;
import org.bukkit.inventory.ItemStack;

import java.util.HashMap;
import java.util.Map;

public class StatsGui {
    
    private final JavaPlugin plugin;
    private final GuiService guiService;
    private final GuiTemplate template;
    private final StatsManager statsManager;
    
    public void open(Player player) {
        PlayerStats stats = statsManager.getStats(player);
        
        Map<String, Object> replacements = new HashMap<>();
        replacements.put("player", player.getName());
        replacements.put("kills", stats.getPlayerKills());
        replacements.put("monster_kills", stats.getMonsterKills());
        replacements.put("deaths", stats.getDeaths());
        replacements.put("playtime", formatPlaytime(stats.getPlaytime()));
        replacements.put("balance", stats.getBalance());
        
        GuiOpenRequest request = new GuiOpenRequest(
            plugin,
            player,
            template,
            replacements,
            source -> null,
            this::renderDynamicSlot,
            new StatsHandler()
        );
        
        guiService.open(request);
    }
    
    private ItemStack renderDynamicSlot(GuiSession session, GuiTemplate.ResolvedSlot slot) {
        String action = slot.definition().action();
        Player player = session.viewer();
        PlayerStats stats = statsManager.getStats(player);
        
        // 可以根据实时数据动态修改显示
        if ("stat_kills".equals(action)) {
            int kd = stats.getDeaths() > 0 ? stats.getPlayerKills() / stats.getDeaths() : stats.getPlayerKills();
            
            return GuiItemBuilder.build(
                "DIAMOND_SWORD",
                new ItemComponentParser.ItemComponents(
                    "<red>击杀统计",
                    true,
                    List.of(
                        "<gray>玩家击杀: <yellow>" + stats.getPlayerKills(),
                        "<gray>怪物击杀: <yellow>" + stats.getMonsterKills(),
                        "<gray>K/D 比率: <gold>" + kd
                    ),
                    null, null, Map.of(), List.of()
                ),
                1,
                session.replacements(),
                source -> null
            );
        }
        
        return null;  // 使用默认渲染
    }
    
    private static class StatsHandler implements GuiSessionHandler {
        @Override
        public void onSlotClick(GuiSession session, InventoryClickEvent event, GuiTemplate.ResolvedSlot slot) {
            if (slot == null) return;
            
            if ("close".equals(slot.definition().action())) {
                session.viewer().closeInventory();
            }
        }
    }
    
    private String formatPlaytime(long seconds) {
        long hours = seconds / 3600;
        long minutes = (seconds % 3600) / 60;
        return hours + "小时 " + minutes + "分钟";
    }
}
```

创建一个支持多种物品源的奖励系统。


### Java 代码

```java
import emaki.jiuwu.craft.corelib.item.*;
import org.bukkit.entity.Player;
import org.bukkit.inventory.ItemStack;

import java.util.*;

public class RewardManager {
    
    private final JavaPlugin plugin;
    private final ItemSourceRegistry registry = ItemSourceRegistry.system();
    private final Map<String, List<RewardItem>> rewardCache = new HashMap<>();
    
    public void loadRewards(ConfigurationSection config) {
        for (String rewardId : config.getKeys(false)) {
            ConfigurationSection rewardSection = config.getConfigurationSection(rewardId);
            List<RewardItem> items = parseRewardItems(rewardSection.getMapList("items"));
            rewardCache.put(rewardId, items);
        }
    }
    
    private List<RewardItem> parseRewardItems(List<Map<?, ?>> itemsConfig) {
        List<RewardItem> items = new ArrayList<>();
        
        for (Map<?, ?> raw : itemsConfig) {
            if (raw instanceof String shorthand) {
                // 解析 "ni-item_id x10" 格式
                items.add(parseShorthand(shorthand));
            } else {
                // 解析完整配置
                String itemId = (String) raw.get("item");
                int amount = raw.containsKey("amount") ? (Integer) raw.get("amount") : 1;
                
                ItemSource source = registry.parseShorthand(itemId);
                if (source != null) {
                    items.add(new RewardItem(source, amount));
                }
            }
        }
        
        return items;
    }
    
    private RewardItem parseShorthand(String shorthand) {
        String[] parts = shorthand.split(" x");
        String itemId = parts[0];
        int amount = parts.length > 1 ? Integer.parseInt(parts[1]) : 1;
        
        ItemSource source = registry.parseShorthand(itemId);
        return source != null ? new RewardItem(source, amount) : null;
    }
    
    public void giveReward(Player player, String rewardId) {
        List<RewardItem> items = rewardCache.get(rewardId);
        if (items == null || items.isEmpty()) {
            player.sendMessage("§c奖励不存在");
            return;
        }
        
        for (RewardItem reward : items) {
            if (reward == null) continue;
            
            ItemStack item = createItem(reward.source(), reward.amount());
            if (item != null) {
                player.getInventory().addItem(item);
            }
        }
        
        player.sendMessage("§a奖励已发放！");
    }
    
    private ItemStack createItem(ItemSource source, int amount) {
        return switch (source.getType()) {
            case VANILLA -> createVanillaItem(source.getIdentifier(), amount);
            case NEIGEITEMS -> createNeigeItemsItem(source.getIdentifier(), amount);
            case CRAFTENGINE -> createCraftEngineItem(source.getIdentifier(), amount);
        };
    }
    
    private ItemStack createVanillaItem(String id, int amount) {
        Material material = Material.matchMaterial(id);
        return material != null ? new ItemStack(material, amount) : null;
    }
    
    private ItemStack createNeigeItemsItem(String id, int amount) {
        return ItemManager.INSTANCE.getItem(id, null, amount);
    }
    
    private ItemStack createCraftEngineItem(String id, int amount) {
        Key key = Key.parse(id);
        CustomItem<?> item = CraftEngineItems.byId(key);
        return item != null ? item.buildItemStack(ItemBuildContext.empty(), amount) : null;
    }
    
    private record RewardItem(ItemSource source, int amount) {}
}
```

## 示例4：表达式计算

### 场景

创建一个动态伤害计算系统。

### 配置文件 (damage_config.yml)

```yaml
damage_formulas:
  base_damage: 50
  
  level_scaling:
    formula: "{base} * (1 + {level} * 0.1)"
  
  critical:
    chance:
      type: "uniform"
      min: 0.05
      max: 0.15
    multiplier:
      type: "gaussian"
      mean: 1.5
      std_dev: 0.2
      min: 1.2
      max: 2.0
  
  elemental_bonus:
    fire:
      type: "constant"
      value: 1.2
    ice:
      type: "skew_normal"
      mean: 1.3
      std_dev: 0.15
      skewness: 0.2
      min: 1.0
      max: 1.8
```

### Java 代码

```java
import emaki.jiuwu.craft.corelib.expression.ExpressionEngine;
import emaki.jiuwu.craft.corelib.math.Randoms;
import org.bukkit.entity.Player;

import java.util.HashMap;
import java.util.Map;

public class DamageCalculator {
    
    private final ConfigurationSection config;
    
    public double calculateDamage(Player attacker, Player target, String element) {
        // 基础伤害
        double baseDamage = config.getDouble("damage_formulas.base_damage");
        
        // 等级缩放
        Map<String, Object> variables = new HashMap<>();
        variables.put("base", baseDamage);
        variables.put("level", attacker.getLevel());
        
        double scaledDamage = ExpressionEngine.evaluate(
            config.getString("damage_formulas.level_scaling.formula"),
            variables
        );
        
        // 暴击判定
        double critChance = ExpressionEngine.evaluateRandomConfig(
            config.get("damage_formulas.critical.chance")
        );
        
        double critMultiplier = 1.0;
        if (Randoms.chance(critChance)) {
            critMultiplier = ExpressionEngine.evaluateRandomConfig(
                config.get("damage_formulas.critical.multiplier")
            );
            attacker.sendMessage("§c暴击！");
        }
        
        // 元素加成
        double elementalBonus = 1.0;
        if (element != null) {
            String path = "damage_formulas.elemental_bonus." + element;
            if (config.contains(path)) {
                elementalBonus = ExpressionEngine.evaluateRandomConfig(config.get(path));
            }
        }
        
        // 最终伤害
        return scaledDamage * critMultiplier * elementalBonus;
    }
}
```

### 场景

创建一个配方解锁系统。

### 配置文件 (recipes.yml)

```yaml

    display_name: "<aqua>钻石剑</aqua>"
    conditions:

      - "%jiuwuforge_crafted_iron_sword% == true"
    condition_type: "all_of"
  
  legendary_sword:
    display_name: "<gold>传说之剑</gold>"
    conditions:
      - "%player_level% >= 50"
      - "%jiuwuforge_reputation% >= 1000"
      - "permission == jiuwuforge.legendary"
      - "%player_rank% == VIP"
    condition_type: "at_least"
    condition_required_count: 3
```

### Java 代码

```java
import emaki.jiuwu.craft.corelib.condition.ConditionEvaluator;
import me.clip.placeholderapi.PlaceholderAPI;
import org.bukkit.entity.Player;

import java.util.List;
import java.util.function.Function;

public class RecipeUnlockManager {
    
    private final Map<String, Recipe> recipes = new HashMap<>();
    
    public boolean canUnlock(Player player, String recipeId) {
        Recipe recipe = recipes.get(recipeId);
        if (recipe == null) return false;
        
        if (recipe.conditions().isEmpty()) {
            return true;
        }
        
        Function<String, String> replacer = text -> {
            // PlaceholderAPI 替换
            text = PlaceholderAPI.setPlaceholders(player, text);
            
            // 自定义变量替换
            text = text.replace("{player_level}", String.valueOf(player.getLevel()));
            
            return text;
        };
        
        return ConditionEvaluator.evaluate(
            recipe.conditions(),
            recipe.conditionType(),
            recipe.conditionRequiredCount(),
            replacer,
            true  // 无效条件视为失败
        );
    }
    
    public List<String> getUnlockableRecipes(Player player) {
        return recipes.keySet().stream()
            .filter(id -> canUnlock(player, id))
            .toList();
    }
    
    public List<String> getLockedRecipes(Player player) {
        return recipes.keySet().stream()
            .filter(id -> !canUnlock(player, id))
            .toList();
    }
}
```

## 示例6：完整插件集成

### 场景

创建一个完整的商店插件，集成所有 CoreLib 功能。

### 主类

```java
import emaki.jiuwu.craft.corelib.gui.*;
import emaki.jiuwu.craft.corelib.item.*;
import emaki.jiuwu.craft.corelib.expression.ExpressionEngine;
import emaki.jiuwu.craft.corelib.condition.ConditionEvaluator;
import emaki.jiuwu.craft.corelib.math.Randoms;
import org.bukkit.plugin.java.JavaPlugin;

public class ShopPlugin extends JavaPlugin implements Listener {
    
    private GuiService guiService;
    private ItemSourceRegistry itemRegistry;
    private ShopManager shopManager;
    
    @Override
    public void onEnable() {
        // 初始化服务
        guiService = new GuiService(this);
        getServer().getPluginManager().registerEvents(guiService, this);
        
        itemRegistry = ItemSourceRegistry.system();
        
        // 加载配置
        saveDefaultConfig();
        loadShopConfig();
        
        // 注册命令
        getCommand("shop").setExecutor(new ShopCommand(this));
        
        getLogger().info("ShopPlugin enabled!");
    }
    
    private void loadShopConfig() {
        ConfigurationSection shops = getConfig().getConfigurationSection("shops");
        shopManager = new ShopManager(this, shops);
    }
    
    public GuiService getGuiService() {
        return guiService;
    }
    
    public ShopManager getShopManager() {
        return shopManager;
    }
}
```

### 商店管理器

```java
public class ShopManager {
    
    private final ShopPlugin plugin;
    private final Map<String, Shop> shops = new HashMap<>();
    
    public void openShop(Player player, String shopId) {
        Shop shop = shops.get(shopId);
        if (shop == null) {
            player.sendMessage("§c商店不存在");
            return;
        }
        
        // 检查条件
        if (!checkConditions(player, shop)) {
            player.sendMessage("§c你不满足进入此商店的条件");
            return;
        }
        
        // 打开 GUI
        new ShopGui(plugin, shop).open(player);
    }
    
    private boolean checkConditions(Player player, Shop shop) {
        if (shop.conditions().isEmpty()) {
            return true;
        }
        
        return ConditionEvaluator.evaluate(
            shop.conditions(),
            shop.conditionType(),
            null,
            text -> PlaceholderAPI.setPlaceholders(player, text),
            true
        );
    }
    
    public boolean purchaseItem(Player player, String shopId, int itemIndex) {
        Shop shop = shops.get(shopId);
        ShopItem item = shop.items().get(itemIndex);
        
        // 计算价格（支持动态定价）
        double price = ExpressionEngine.evaluate(
            item.priceFormula(),
            Map.of("player_level", player.getLevel())
        );
        
        // 检查余额
        if (!hasEnoughMoney(player, price)) {
            player.sendMessage("§c金币不足！需要 " + price);
            return false;
        }
        
        // 扣款
        withdrawMoney(player, price);
        
        // 给予物品
        ItemStack itemStack = createItem(item.source(), item.amount());
        player.getInventory().addItem(itemStack);
        
        player.sendMessage("§a购买成功！");
        return true;
    }
    
    private ItemStack createItem(ItemSource source, int amount) {
        return switch (source.getType()) {
            case VANILLA -> new ItemStack(Material.matchMaterial(source.getIdentifier()), amount);
            case NEIGEITEMS -> ItemManager.INSTANCE.getItem(source.getIdentifier(), null, amount);
            case CRAFTENGINE -> CraftEngineItems.byId(Key.parse(source.getIdentifier()))
                .buildItemStack(ItemBuildContext.empty(), amount);
        };
    }
}
```

## 总结

以上示例展示了 Emaki CoreLib 的主要功能和使用方式：

1. **GUI 系统**：模板化设计、动态渲染、事件处理
2. **物品源系统**：统一接口、多源支持、易于扩展
3. **表达式引擎**：数学计算、随机分布、变量替换
4. **条件评估器**：灵活判断、多种组合模式

通过合理组合这些模块，可以快速构建功能丰富的 Minecraft 插件。
